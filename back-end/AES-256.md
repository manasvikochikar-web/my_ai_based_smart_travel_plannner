from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend
from cryptography.hazmat.primitives import padding
import os

def encrypt(plain_text, key):
    # Generate a random 16-byte IV
    iv = os.urandom(16)
    cipher = Cipher(algorithms.AES(key), modes.CBC(iv), backend=default_backend())
    encryptor = cipher.encryptor()

    # Pad the data to 16-byte blocks (AES standard)
    padder = padding.PKCS7(128).padder()
    padded_data = padder.update(plain_text.encode()) + padder.finalize()

    # Encrypt
    cipher_text = encryptor.update(padded_data) + encryptor.finalize()
    return iv, cipher_text

def decrypt(iv, cipher_text, key):
    cipher = Cipher(algorithms.AES(key), modes.CBC(iv), backend=default_backend())
    decryptor = cipher.decryptor()

    # Decrypt
    padded_data = decryptor.update(cipher_text) + decryptor.finalize()

    # Remove padding
    unpadder = padding.PKCS7(128).unpadder()
    data = unpadder.update(padded_data) + unpadder.finalize()
    return data.decode()

# --- Main Interaction ---
if __name__ == "__main__":
    # 1. Setup a secure 256-bit key
    key = os.urandom(32) 
    
    # 2. Get user input
    user_msg = input("Enter the secret message you want to encrypt: ")

    # 3. Run Encryption
    iv, encrypted_msg = encrypt(user_msg, key)
    
    print("\n--- Results ---")
    print(f"Original:  {user_msg}")
    print(f"Key (Hex): {key.hex()}")
    print(f"IV (Hex):  {iv.hex()}")
    print(f"Encrypted: {encrypted_msg.hex()}")

    # 4. Run Decryption to verify
    decrypted_msg = decrypt(iv, encrypted_msg, key)
    print(f"Decrypted: {decrypted_msg}")
